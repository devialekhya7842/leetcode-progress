# Definition for singly-linked list.
# class ListNode(object):
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution(object):
    def isPalindrome(self, head):
        """
        :type head: Optional[ListNode]
        :rtype: bool
        """
        if head == None or head.next == None:
            return True
        def reverse_LL(r_head):
            if r_head == None or r_head.next== None:
                return r_head
            new_r_head=reverse_LL(r_head.next)
            r_head.next.next=r_head
            r_head.next=None
            return new_r_head
             

        slow=head
        fast=head
        while(fast.next != None and fast.next.next != None):
            slow=slow.next
            fast=fast.next.next
        new_head=reverse_LL(slow.next)
        
        first,second=head,new_head
        while(second != None):
            if first.val != second.val:
                reverse_LL(new_head)
                return False
            first =first.next
            second=second.next
            
        reverse_LL(new_head)
        return True
