#day14: single number(LC)
#link: https://leetcode.com/problems/single-number/?envType=problem-list-v2&envId=array
#date: 20-12-2025


class Solution(object):
    def singleNumber(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        d={}
        for i in nums:
            d[i]=d.get(i, 0)+1
        for k,v in d.items():
            if v==1:
                return k
