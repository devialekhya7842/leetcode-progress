#problem:Array Subset- Given two arrays a[] and b[], your task is to determine whether b[] is a subset of a[].
#10-11-2025


#User function Template for python3

class Solution:
    #Function to check if a is a subset of b.
    def isSubset(self, a, b):
        # Your code here
        d1={}
        for i in a:
            d1[i] =d1.get(i,0)+1
        for x in b:
            if x not in d1 or d1[x] == 0:
                return False
            d1[x] -= 1

        return True
