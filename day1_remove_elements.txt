#remove element
#date:10-nov-2025
#https://leetcode.com/problems/remove-element/

class Solution(object):
    def removeElement(self, nums, val):
        """
        :type nums: List[int]
        :type val: int
        :rtype: int
        """
        l=[]
        for i in nums:
            if i != val:
                l.append(i)
        for i in range(len(l)):
            nums[i]=l[i]
        return len(l)
