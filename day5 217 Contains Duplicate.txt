#day5 217. Contains Duplicate
#link https://leetcode.com/problems/contains-duplicate/description/
15-11-2025

class Solution(object):
    def containsDuplicate(self, nums):
        """
        :type nums: List[int]
        :rtype: bool
        """
        d={}
        for i in nums:
            if i in d:
                d[i]+=1
                if d[i]>1:
                    return True
            else:
                d[i]=1

        return False
