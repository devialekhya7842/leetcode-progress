#day6:268. Missing Number
#link: https://leetcode.com/problems/missing-number/description/
#date: 16-11-2025


class Solution(object):
    def missingNumber(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        for i in range(1,len(nums)+1):
            if i not in nums:
                return i
        return 0
